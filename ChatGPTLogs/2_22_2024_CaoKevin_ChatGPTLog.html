public void drawDShape(Graphics2D g, Point start, Point end) {
    g.setColor(Color.RED);
    g.setStroke(new BasicStroke(2));

    int x = start.x - DOOR_SIZE / 2;
    int y = start.y - DOOR_SIZE / 2;

    // Draw the semicircle
    g.drawArc(x, y, DOOR_SIZE, DOOR_SIZE, 90, 180);

    // Calculate the center of the semicircle
    int centerX = x + DOOR_SIZE / 2;
    int centerY = y + DOOR_SIZE / 2;

    // Draw a line from one endpoint of the semicircle to the other endpoint

}
